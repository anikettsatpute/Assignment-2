% Lists frame
\begin{frame}{Solution :}
$\bar{x}$ = 6 , $\bar{y}$ = 8\\
$\sigma_{x}$ = 4 , $\sigma_{y}$ = 6\\ 
r = $\frac{2}{3}$\\
\begin{align}
b_{yx} &= r.\frac{\sigma_{y}}{\sigma_{x}}
        = \frac{2}{3}.\frac{6}{4}
        = 1\\
b_{xy} &= r.\frac{\sigma_{x}}{\sigma_{y}}
        = \frac{2}{3}.\frac{4}{6}
        = \frac{4}{9}
\end{align}

\end{frame}

\end{align*}
